96int
97list_add(ps_list p_list,
98 void* pv_data) {
99 if ( p_list != NULL ) {
100 if ( p_list->is_sorted == false ) {
101 if ( p_list->p_first == NULL ) {
102 p_list->p_first = calloc(1, sizeof(s_list_data)) ;
103 p_list->p_first->pv_data = pv_data ;
104 p_list->p_last = p_list->p_first ;
105 } else {
106 ps_list_data p_current = calloc(1, sizeof(s_list_data)) ;
107 p_current->pv_data = pv_data ;
108 p_current->p_prev = p_list->p_last ;
109 p_list->p_last->p_next = p_current ;
110 p_list->p_last = p_current ;
111 }
112 p_list->count++;
113 } else {
114 if ( p_list->pf_compare != NULL ) {
115 if ( p_list->p_first == NULL ) {
116 p_list->p_first = calloc(1, sizeof(s_list_data)) ;
117 p_list->p_first->pv_data = pv_data ;
118 p_list->p_last = p_list->p_first ;
119 } else {
120 if ( p_list->pf_compare(pv_data, p_list->p_first->pv_data) < 0 ) {
121 // insert before first
122 ps_list_data p_new = calloc(1, sizeof(s_list_data)) ;
123 p_new->pv_data = pv_data ;
124 p_new->p_next = p_list->p_first ;
125 p_list->p_first->p_prev = p_new ;
126 p_new->p_prev = NULL ;
127 p_list->p_first = p_new ;
128 } else if ( p_list->pf_compare(pv_data, p_list->p_last->pv_data) > 0 ) {
129 // insert after last
130 ps_list_data p_new = calloc(1, sizeof(s_list_data)) ;
131 p_new->pv_data = pv_data ;
132 p_new->p_prev = p_list->p_last;
133 p_new->p_next = NULL ;
134 p_list->p_last->p_next = p_new ;
135 p_list->p_last = p_new ;
136 } else {
137 // search insert cellule
138 ps_list_data p_current = p_list->p_first;
139 unsigned int index = 0 ;
140 while ( p_list->pf_compare(pv_data, p_current->pv_data) >= 0 &&
141 index < p_list->count ) {
142 p_current = p_current->p_next ;
143 index++;
144 }
145 // insert before p_current
146 ps_list_data p_new = calloc(1, sizeof(s_list_data)) ;
147 p_new->pv_data = pv_data ;
148 p_new->p_next = p_current ;
149 p_new->p_prev = p_current->p_prev ;
150 p_new->p_prev->p_next = p_new ;
151 p_current->p_prev = p_new ;
152 }
153 }
154 p_list->count++;
155 }
156 }
157 return p_list->count;
158 }
159 return 0;
160}
