185void
186hashtable_put(ps_hashtable p_hashtable, void* pv_key, size_t key_size, void* pv_value) {
187 if ( p_hashtable->pf_comp != NULL ) {
188 unsigned int ui_hash = p_hashtable->pf_hash(pv_key, key_size) ;
189 unsigned int ui_hash_index = ui_hash % p_hashtable->v_entries.capacity ;
190 void* pv_vector_values = hashtable_vector_element_at(&(p_hashtable->v_entries), ui_hash_index) ;
191 ps_vector p_array_of_values = NULL;
192
193 if ( pv_vector_values == NULL ) {
194 p_array_of_values = calloc(sizeof(s_vector), 1);
195 vector_init(p_array_of_values, 10);
196 ps_hash_entry p_hash_entry = calloc(sizeof(s_hash_entry),1);
197 p_hash_entry->pv_key = pv_key ;
198 p_hash_entry->key_size = key_size ;
199 p_hash_entry->pv_value = pv_value ;
200 vector_add_element(p_array_of_values, p_hash_entry);
201 hashtable_vector_set_element_at(&(p_hashtable->v_entries), p_array_of_values, ui_hash_index);
202 p_hashtable->nb_elements++;
203 } else {
204 p_array_of_values = (ps_vector)pv_vector_values;
205 ps_hash_entry p_hash_entry = calloc(sizeof(s_hash_entry),1);
206 p_hash_entry->pv_key = pv_key ;
207 p_hash_entry->key_size = key_size ;
208 p_hash_entry->pv_value = pv_value ;
209 long index_of_existing_element = hashtable_vector_index_of(p_array_of_values, p_hash_entry, p_hashtable->pf_comp);
210 if ( index_of_existing_element >= 0 ) {
211 ps_hash_entry old_hash_entry = vector_element_at(p_array_of_values, index_of_existing_element);
212 if ( old_hash_entry != NULL ) {
213 free(old_hash_entry);
214 }
215 vector_set_element_at(p_array_of_values, p_hash_entry, index_of_existing_element);
216 } else {
217 vector_add_element(p_array_of_values, p_hash_entry);
218 p_hashtable->nb_elements++;
219 }
220 }
221
222 if ( p_hashtable->nb_elements >= p_hashtable->threshold ) {
223 unsigned int new_capacity = ((p_hashtable->v_entries.capacity) << 1) + 1 ;
224 printf("rehash is needed nb_elements(%u) threshold(%u) old_capacity(%u) new_capacity(%u)\n",
225 p_hashtable->nb_elements,
226 p_hashtable->threshold,
227 p_hashtable->v_entries.capacity,
228 new_capacity);
229 hashtable_rehash(p_hashtable, new_capacity);
230 }
231 }
